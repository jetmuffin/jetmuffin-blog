之后发现加了一层黑色的遮罩后，弹出框必定在遮罩上方，这样其实只要给黑色遮罩绑定事件即可。

代码：

`// 关闭登录框

var close_login_box= function (){

$('#black-layer').fadeOut();

$('#login-pop').fadeOut();

};

// 登录按钮点击事件

$(‘#login-btn’).click(function(){

$('#black-layer').fadeIn();

$('#login-pop').fadeIn();

return false;

});

// 点击黑色区域直接关闭登陆框

$(‘#black-layer’).click(close_login_box);`
